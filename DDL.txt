create table if not exists ingrediente(
	id_ingrediente serial
	,descricao varchar(90)
	,unit varchar(2)
	,preco double
	,volume double
	,preco_unit double
	,qtd_reserva double
	,qtd_necessaria double
);

create table if not exists receita(
	id_ingrediente int
	,qtd_necessaria double
	,receita varchar(80)
	,cooking boolean default false
);
